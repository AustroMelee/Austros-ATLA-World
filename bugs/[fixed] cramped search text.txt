export default function SearchBar({ query, onQueryChange, suggestion, textColor }: SearchBarProps) {
  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Tab' && suggestion && query) {
      e.preventDefault();
      onQueryChange(query + suggestion);
    }
  };

  const showSuggestion = suggestion && query.length > 0;

  return (
    <div className="relative w-full max-w-2xl mx-auto">
      {/*
        LAYER 1: The Visuals ("Fake Input")
        This div is styled to look like an input and is not interactive.
      */}
      <div
        className="w-full h-[52px] px-5 py-3 flex items-center justify-between rounded-lg bg-slate-800 font-medium text-base border-2 border-slate-700 pointer-events-none"
        aria-hidden="true"
      >
        {/* This container ensures text flows left-to-right and doesn't wrap. */}
        <div className="flex-1 flex items-center whitespace-nowrap overflow-hidden">
          <span style={{ color: textColor }}>{query}</span>
          {showSuggestion && (
            <span className="text-slate-400/50 text-ellipsis overflow-hidden">
              {suggestion}
            </span>
          )}
        </div>

        {/* Tab keycap, only shown when there's a suggestion */}
        {showSuggestion && (
          <span className="ml-4 flex-shrink-0 text-xs font-semibold uppercase text-slate-400 bg-slate-900/75 border border-slate-700 rounded-md px-2 py-0.5">
            Tab
          </span>
        )}
      </div>

      {/*
        LAYER 2: The Interaction (Real, Invisible Input)
        Layered on top to capture typing.
      */}
      <input
        className="absolute inset-0 w-full h-full px-5 py-3 rounded-lg bg-transparent text-transparent font-medium text-base border-2 border-transparent focus:outline-none focus:ring-2 focus:ring-blue-400"
        style={{ caretColor: textColor }}
        value={query}
        onChange={e => onQueryChange(e.target.value)}
        onKeyDown={handleKeyDown}
        placeholder="Search for 'Toph' or 'Fire Flakes'..."
        aria-label="Search encyclopedia"
        autoComplete="off"
        type="text"
      />
    </div>
  );
}

Why This Final Version Will Work

Corrected Text Flow: The key change is the new inner <div className="flex-1 flex items-center whitespace-nowrap overflow-hidden">. This container forces the query and suggestion spans to render sequentially as a single line (whitespace-nowrap). This physically prevents them from ever overlapping.

Overflow Handling: The overflow-hidden and text-ellipsis on the suggestion span will gracefully truncate the suggestion if the combined text is too long, preventing it from pushing the "Tab" keycap out of view.

Robust Spacing: The parent div uses justify-between, and the "Tab" keycap has flex-shrink-0, which guarantees that the text block and the keycap are spaced apart reliably.

This is the industry-standard, robust way to build a custom input with this kind of functionality. It is no longer fighting the browser's rendering quirks but is instead using a predictable and clean layout structure.

**Solution to Text Cramping in the SearchBar Autosuggestion:**

The text cramping (overlap or crowding between the user's query, the suggestion, and the "TAB" keycap) was fixed by making the following key changes:

---

### 1. **Proper Flexbox Layout with Shrinkable Container**
- The text container (holding the query and suggestion) was given `min-w-0` and `overflow-hidden` inside a flex row.
- This allows the suggestion text to shrink and ellipsis (`...`) when space is tight, instead of pushing into the keycap or overflowing.

### 2. **Margin Between Suggestion and Keycap**
- A right margin (`mr-4` or, in the final version, a left margin `ml-2` on the keycap) was added to ensure there is always space between the suggestion and the "TAB" keycap, preventing them from touching or overlapping.

### 3. **Keycap Placement**
- The "TAB" keycap was moved to be immediately after the suggestion (not at the far right), so the suggestion and keycap are always visually connected and never overlap.

---

**Key Code Snippet:**
```tsx
<div className="flex items-center whitespace-nowrap overflow-hidden min-w-0">
  <span style={{ color: textColor }}>{query}</span>
  {showSuggestion && (
    <>
      <span className="text-slate-400/50 text-ellipsis overflow-hidden ml-1">
        {suggestion}
      </span>
      <span className="flex-shrink-0 text-xs font-semibold uppercase text-slate-400 bg-slate-900/75 border border-slate-700 rounded-md px-2 py-0.5 ml-2">
        Tab
      </span>
    </>
  )}
</div>
```

---

**Summary:**  
- **min-w-0** and **overflow-hidden** allow the suggestion to shrink and ellipsis, preventing cramping.
- **Margin** between suggestion and keycap ensures clear separation.
- **Keycap placement** right after the suggestion keeps the UI clean and readable.

This combination ensures the query, suggestion, and keycap never overlap or cramp, regardless of input length.
